## 1. 异步运行机制

JavaScript 是单线程的，通过事件循环处理异步任务。

**执行顺序：** 同步代码 → 微任务（Promise.then）→ 宏任务（setTimeout）

```js
console.log('同步代码1');
setTimeout(() => console.log('宏任务: setTimeout'), 0);
Promise.resolve()
  .then(() => console.log('微任务: Promise.then'))
  .then(() => console.log('微任务: 链式Promise'));
console.log('同步代码2');

// 输出：同步代码1 → 同步代码2 → 微任务: Promise.then → 微任务: 链式Promise → 宏任务: setTimeout
```

## 2. AI 对话技术：Fetch + SSE

```js
class AIStreamChat {
  constructor(endpoint) {
    this.endpoint = endpoint;
    this.eventSource = null;
  }

  connect(onMessage, onError) {
    this.eventSource = new EventSource(this.endpoint);
    this.eventSource.onmessage = (event) => {
      const data = JSON.parse(event.data);
      onMessage(data);
    };
    this.eventSource.onerror = onError;
  }

  async sendMessage(message) {
    const response = await fetch('/api/chat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ message })
    });
    return await response.json();
  }

  disconnect() {
    this.eventSource?.close();
    this.eventSource = null;
  }
}
```

## 3. 会话中断处理机制

**中断表现：** 网络断开、服务器超时、连接异常关闭、用户关闭页面

```js
class SessionManager {
  constructor() {
    this.maxReconnectAttempts = 5;
    this.currentAttempts = 0;
  }

  initNetworkListeners() {
    window.addEventListener('online', () => this.handleReconnect());
    window.addEventListener('offline', () => this.handleDisconnect());
  }

  handleReconnect() {
    if (this.currentAttempts < this.maxReconnectAttempts) {
      this.currentAttempts++;
      // 指数退避策略
      setTimeout(() => this.attemptReconnect(),
        Math.pow(2, this.currentAttempts) * 1000);
    }
  }

  startHeartbeat() {
    this.heartbeatTimer = setInterval(async () => {
      try {
        await fetch('/api/ping');
      } catch {
        this.handleDisconnect();
      }
    }, 30000);
  }
}
```

## 4. npm vs pnpm 区别

| 维度 | npm | pnpm |
|------|-----|------|
| 依赖管理 | 扁平化，可能有幽灵依赖 | 硬链接，严格依赖隔离 |
| 存储机制 | 每个项目独立存储 | 全局存储 + 硬链接 |
| 安装速度 | 较慢 | 更快（复用已下载的包） |
| 磁盘空间 | 占用大 | 节省 50-90% |
| 幽灵依赖 | 可能访问未声明依赖 | 严格控制，避免幽灵依赖 |
| 兼容性 | 最好 | 可能存在某些包的兼容性问题 |

## 5. 手写发布订阅模式

```js
class PubSub {
  constructor() {
    this.subscribers = {};
  }

  subscribe(event, callback) {
    if (!this.subscribers[event]) this.subscribers[event] = [];
    this.subscribers[event].push(callback);
    return () => this.unsubscribe(event, callback); // 返回取消订阅函数
  }

  publish(event, data) {
    (this.subscribers[event] || []).forEach(cb => {
      try { cb(data); } catch (e) { console.error(e); }
    });
  }

  unsubscribe(event, callback) {
    if (!this.subscribers[event]) return;
    this.subscribers[event] = this.subscribers[event].filter(s => s !== callback);
  }

  once(event, callback) {
    const onceCallback = (data) => {
      callback(data);
      this.unsubscribe(event, onceCallback);
    };
    this.subscribe(event, onceCallback);
  }
}

// 使用示例
const pubsub = new PubSub();
pubsub.subscribe('userLogin', (user) => console.log('用户登录:', user.name));
pubsub.publish('userLogin', { id: 1, name: '张三' });
```

## 6. 手写防抖函数（完整版）

```js
function debounce(func, delay, immediate = false) {
  let timeoutId = null;

  const debounced = function (...args) {
    clearTimeout(timeoutId);

    if (immediate) {
      const callNow = !timeoutId;
      timeoutId = setTimeout(() => { timeoutId = null; }, delay);
      if (callNow) func.apply(this, args);
    } else {
      timeoutId = setTimeout(() => {
        func.apply(this, args);
      }, delay);
    }
  };

  debounced.cancel = function () {
    clearTimeout(timeoutId);
    timeoutId = null;
  };

  return debounced;
}

// 使用
const searchHandler = debounce((query) => console.log('搜索:', query), 300);
searchHandler('a');
searchHandler('ab');
searchHandler('abc'); // 只有这次会执行
```
