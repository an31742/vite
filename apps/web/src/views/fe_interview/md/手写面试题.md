## 1. 递归生成不重复随机数组

生成长度为 5、元素在 2-32 之间不重复的数组：

```js
function buildArray(arr, length, min, max) {
  if (arr.length > length) return arr;
  let num = Math.floor(Math.random() * (max - min + 1) + min);
  if (!arr.includes(num)) arr.push(num);
  return buildArray(arr, length, min, max);
}

console.log(buildArray([], 5, 2, 32)); // [x, x, x, x, x]
```

## 3. 翻转字符串大小写

```js
function toggleCase(str) {
  let arr = [];
  for (let item of str) {
    item = item === item.toUpperCase() ? item.toLowerCase() : item.toUpperCase();
    arr.push(item);
  }
  return arr.join("");
}

toggleCase("aBcD"); // "AbCd"
```

## 4. 统计字符串出现次数

```js
// 方法1：match + replace 循环计数
function strCount(str, target) {
  let count = 0;
  if (!target) return count;
  while (str.match(target)) {
    str = str.replace(target, "");
    count++;
  }
  return count;
}

// 方法2：split 分割
"abcdabcdabcd".split('a').length - 1; // 3
```

## 5. 手写 Promise.retry

```js
Promise.retry = function (fn, times) {
  return new Promise((resolve, reject) => {
    fn()
      .then(resolve)
      .catch((err) => {
        if (times > 0) {
          Promise.retry(fn, times - 1).then(resolve).catch(reject);
        } else {
          reject(err);
        }
      });
  });
};

// 使用
Promise.retry(mockRequest, 3)
  .then(() => console.log("成功"))
  .catch((err) => console.log("重试3次失败", err));
```

**递归 vs 循环：**

| 方面 | 循环 | 递归 |
|------|------|------|
| 内存 | 常数空间 | O(n) 调用栈 |
| 性能 | 更快 | 有函数调用开销 |
| 可读性 | 简单重复场景直观 | 树形/分治问题优雅 |
| 风险 | 无限循环 | 栈溢出 |

## 6. 手写防抖（debounce）

多次触发只执行最后一次：

```js
function debounce(fn, delay) {
  let timer = null;
  return function (...args) {
    const context = this;
    if (timer) clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(context, args);
    }, delay);
  };
}
```

## 7. 手写节流（throttle）

每隔一定时间触发一次：

```js
// 时间戳版
function throttle(fn, delay) {
  let lastTime = 0;
  return function (...args) {
    const nowTime = Date.now();
    if (nowTime - lastTime > delay) {
      lastTime = nowTime;
      return fn.apply(this, args);
    }
  };
}

// 定时器版
function throttle2(fn, delay) {
  let timer = null;
  return function (...args) {
    if (timer) return;
    timer = setTimeout(() => {
      fn.apply(this, args);
      timer = null;
    }, delay);
  };
}
```
