function onScroll(e) {
  const scrollTop = e.target.scrollTop  // 滚动了多少px

  // 从第几条开始渲染
  const startIndex = Math.floor(scrollTop / itemHeight)
  // 到第几条结束（多渲染5条作为缓冲，防止白屏）
  const endIndex = startIndex + visibleCount + 5

  // 截取这个范围的数据
  visibleList = list.slice(startIndex, endIndex)

  // 列表整体下移，对齐滚动位置
  offsetY = startIndex * itemHeight
}

async function startStream(prompt, retryCount = 0) {
  const MAX_RETRY = 3
  const controller = new AbortController()

  try {
    const response = await fetch('/api/chat', {
      method: 'POST',
      body: JSON.stringify({ prompt }),
      signal: controller.signal  // 用于手动取消
    })

    const reader = response.body.getReader()
    const decoder = new TextDecoder()

    while (true) {
      const { done, value } = await reader.read()

      if (done) break  // 正常结束

      content.value += decoder.decode(value)
    }

  } catch (err) {
    if (err.name === 'AbortError') {
      // 用户主动取消，不重试
      return
    }

    // 网络中断，重试
    if (retryCount < MAX_RETRY) {
      status.value = `连接中断，第${retryCount + 1}次重试...`
      setTimeout(() => {
        startStream(prompt, retryCount + 1)  // 递归重试
      }, 1000 * (retryCount + 1))  // 间隔递增 1s 2s 3s
    } else {
      status.value = '连接失败，请重试'
    }
  }
}

// 用户手动取消（比如点击停止按钮）
function stopStream() {
  controller.abort()
}

let receivedContent = ''  // 记录已收到的内容

async function startStream(prompt) {
  const response = await fetch('/api/chat', {
    method: 'POST',
    body: JSON.stringify({
      prompt,
      offset: receivedContent.length  // 告诉服务端已收到多少字符
    })
  })

  const reader = response.body.getReader()
  while (true) {
    const { done, value } = await reader.read()
    if (done) break
    const chunk = decoder.decode(value)
    receivedContent += chunk      // 累积记录
    content.value = receivedContent  // 展示
  }
}

## ref和reactive 有什么区别
ref 什么类型都能包，访问要加.value，解构不丢响应式；reactive 只能包对象，访问不用.value，但解构会丢响应式需要toRefs；基本类型必须用ref，复杂对象两个都行看个人习惯


## 为什么ref是.value 而reactive 不是.value  为什么ref要包一层
基本类型没法被 Proxy 监听
↓
ref 把它包进 { value: xxx } 变成对象
↓

↓
所以访问要加 .value

reactive 本来就是对象，Proxy 直接监听，不需要包一层


## 计算属性的原理是什么为什么有缓存
计算属性内部用 dirty 标记控制是否重新计算，第一次访问执行getter并收集依赖，结果缓存起来dirty置为false；依赖变了只是把dirty置为true，等下次访问才重新计算；依赖没变dirty是false直接返回缓存，所以不管访问多少次都只计算一次。

```
// computed 内部简化实现
function computed(getter) {
  let dirty = true      // true = 需要重新计算
  let cachedValue       // 缓存的结果

  const effect = new ReactiveEffect(getter, () => {
    // 依赖变了，把dirty标记为true
    // 但不立刻重新计算，等下次访问再算（懒执行）
    dirty = true
  })

  return {
    get value() {
      if (dirty) {
        dirty = false
        cachedValue = effect.run()  // 重新计算，收集依赖
      }
      return cachedValue            // 返回缓存
    }
  }
}

```

## watch 和 watchEffect 有什么区别
watch 手动指定依赖，默认不立刻执行，能拿新旧值，适合需要对比变化的场景；watchEffect 自动收集依赖，初始化立刻执行，拿不到旧值，适合多个依赖联动触发同一逻辑的场景。


```
// 场景1：监听路由参数变化请求数据，需要对比新旧值
watch(() => route.params.id, (newId, oldId) => {
  if (newId !== oldId) fetchData(newId)
})

// 场景2：多个筛选条件变化就重新请求，用watchEffect更简洁
watchEffect(() => {
  // 用到了哪些就监听哪些，不用一个个列出来
  fetchList({
    keyword: keyword.value,
    page: page.value,
    status: status.value
  })
})

```

## 如何在一个对象里面判断所有的数据类型
用 Object.prototype.toString.call(value) 判断最准，能识别所有类型包括 null/array/date/map；typeof 会把 null 和数组都判断成 object；遍历对象用 for...in + hasOwnProperty 过滤原型链属性，对每个值调用 getType 即可

```
type TypeName =
  | 'string' | 'number' | 'boolean'
  | 'null' | 'undefined' | 'array'
  | 'object' | 'function' | 'date'
  | 'map' | 'set' | 'symbol'

function getType(value: unknown): TypeName {
  return Object.prototype.toString.call(value).slice(8, -1).toLowerCase() as TypeName
}

// 遍历对象
function checkObjectTypes<T extends Record<string, unknown>>(obj: T) {
  return Object.fromEntries(
    Object.entries(obj).map(([key, val]) => [key, getType(val)])
  ) as Record<keyof T, TypeName>
}
```

## 如果页面卡顿有哪些排查思路  
Performance 面板录制卡顿过程
↓
看火焰图找耗时长的函数
↓
看 FPS 是否低于 60
↓
看 Long Task（超过50ms的任务）


先用 Performance 面板录制定位是 JS 执行慢、重排重绘多、还是内存泄漏；JS 慢用时间切片；DOM 操作多用读写分离或 transform；列表长用虚拟滚动；事件频繁用防抖节流；组件多余渲染用 v-memo；内存泄漏检查事件监听和定时器有没有在 onUnmounted 清除。


```
1. JS 执行时间太长
// 问题：大量数据同步处理，阻塞主线程
const result = bigList.map(item => heavyCalculate(item))  // 卡死页面

// 解决：时间切片，分批处理
function processInChunks(list, chunkSize = 100) {
  let index = 0

  function processChunk() {
    const chunk = list.slice(index, index + chunkSize)
    chunk.forEach(item => heavyCalculate(item))
    index += chunkSize

    if (index < list.length) {
      requestAnimationFrame(processChunk)  // 下一帧继续，不阻塞渲染
    }
  }

  processChunk()
}
2. 频繁触发重排重绘
// 问题：循环里读写DOM，触发强制重排
for (let i = 0; i < 100; i++) {
  const height = el.offsetHeight    // 读，触发重排
  el.style.height = height + 1 + 'px'  // 写，触发重排
}
// 100次重排，页面卡死

// 解决：读写分离，批量操作
const height = el.offsetHeight      // 先统一读
for (let i = 0; i < 100; i++) {
  el.style.height = height + i + 'px'  // 再统一写
}

// 更好的方案：用 transform 代替改 height
// transform 不触发重排，走GPU合成层
el.style.transform = `scaleY(${ratio})`


3. 列表数据量太大

// 问题：渲染10000条DOM节点
<div v-for="item in list10000">  // 卡

// 解决：虚拟滚动，只渲染可视区域
<VirtualList :list="list10000" :itemHeight="40" />

4. 事件监听没有防抖节流

// 问题：滚动/输入触发大量计算
window.addEventListener('scroll', heavyFn)   // 每帧触发
input.addEventListener('input', searchFn)    // 每次输入触发

// 解决：防抖节流
import { debounce, throttle } from 'lodash'

window.addEventListener('scroll', throttle(heavyFn, 100))  // 100ms一次
input.addEventListener('input', debounce(searchFn, 300))   // 停止输入300ms后触发


5. 内存泄漏导致越用越卡

// 问题：组件销毁了但监听没清除，内存一直涨
onMounted(() => {
  window.addEventListener('resize', handleResize)
  timer = setInterval(fn, 1000)
})

// 解决：组件销毁时清除
onUnmounted(() => {
  window.removeEventListener('resize', handleResize)
  clearInterval(timer)
})

6. Vue 组件过度渲染

// 问题：父组件更新，所有子组件跟着重渲染
<ChildComponent v-for="item in list" />

// 解决：加key + 用memo缓存
// Vue3 用 v-memo 跳过不必要的渲染
<div v-for="item in list" :key="item.id" v-memo="[item.id, item.status]">
  // 只有 id 或 status 变了才重新渲染
</div>


```
1. Performance 录制 → 找 Long Task 和耗时函数
2. 看 FPS 面板 → 低于60就有掉帧
3. 看 Memory 面板 → 内存持续增长说明有泄漏
4. 看 Network 面板 → 接口慢导致的卡顿
